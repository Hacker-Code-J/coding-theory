\section{Definition of Syzygy}
\begin{observation}
The linear dependence of vectors \(\textbf{v}_1,\dots,\textbf{v}_r\in V\) is expressed by %an equation
\[
c_1\textbf{v}_1+\cdots+c_r\textbf{v}_r=\textbf{0}
\]
for some scalars \(c_1,\dots,c_r\in\F\).

In commutative algebra, one studies an analogous notion for elements of a ring, but now the coefficients are allowed to lie in the ring \(R\) itself.

\medskip

\noindent
\textbf{\color{red}Question.}\; 
Let $R$ be a ring.
Suppose \(f_1,\dots,f_r\in R\).  
Can we find elements \(a_1,\dots,a_r\in R\) such that
\[
a_1f_1+\cdots+a_rf_r=0\in R?
\]
If yes, then the tuple \((a_1,\dots,a_r)\) records a relation among the \(f_i\).  
That relation is called a \emph{syzygy}.

\end{observation}

\medskip

\defbox[Syzygy]{\begin{definition}
Let \(R\) be a commutative ring and let \(f_1,\dots,f_r \in R\). A \textbf{syzygy} among \(f_1,\dots,f_r\) is a tuple
\[(a_1,\dots,a_r)\in R^r\quad\text{such that}\quad
a_1f_1+\cdots+a_rf_r=0.\]
\end{definition}}
\begin{remark}
This is richer than linear algebra because coefficients are no longer just constants; they may be polynomials or other ring elements.
%So a syzygy is a relation among the elements \(f_1,\dots,f_r\), where the coefficients \(a_i\) are taken from the ring \(R\).
\end{remark}

\medskip

\begin{example}
Let \[
f_1=x,\quad f_2=y\quad\quad\text{where}\; f_1,f_2\in\C[x,y].
\]
A syzygy of \(f_1,f_2\) is a pair \((a_1,a_2)\in \C[x,y]^2\) such that
\[
a_1x+a_2y=0.
\]
The pair $(-y,x)$ is a syzygy of \(x\) and \(y\), because
\[
(-y)x+xy=-xy+xy=0.
\]
\end{example}

\section{Syzygies as Kernels}
\begin{observation}
%[The set of all linear dependence relations among vectors forms a vector space.]
Suppose $
\textbf{v}_1,\textbf{v}_2,\dots,\textbf{v}_r\in V
$ where $V$ is a vector space over $\F$. Define \[
L:=\set{(c_1,c_2,\dots,c_r)\in\F^r:c_1\textbf{v}_1+c_2\textbf{v}_2+\cdots+c_r\textbf{v}_r=\textbf{0}}\subseteq\F^r.
\] Then $L$ be a vector space: \begin{enumerate}[(i)]
\item (Vector Addition)\;. Let $(c_1,c_2,\dots,c_r),(d_1,d_2,\dots,d_r)\in L$. Since \[
c_1\textbf{v}_1+c_2\textbf{v}_2+\cdots+c_r\textbf{v}_r=\textbf{0},\quad d_1\textbf{v}_1+d_2\textbf{v}_2+\cdots+d_r\textbf{v}_r=\textbf{0}
\] we have \[
(c_1+d_1)\textbf{v}_1+(c_2+d_2)\textbf{v}_2+\cdots+(c_r+d_r)\textbf{v}_r=\textbf{0}.
\]
\item (Scalar Multiplication).\; Let $\lambda\in \F$. Then \[
(\lambda c_1)\textbf{v}_1+(\lambda c_2)\textbf{v}_2+\cdots+(\lambda c_r)\textbf{v}_r=\lambda(c_1\textbf{v}_1+c_2\textbf{v}_2+\cdots+c_r\textbf{v}_r)=\lambda\textbf{0}=\textbf{0}.
\]
\end{enumerate}
%Hence the set of all relations is a vector space.
\end{observation}

\defbox[Syzygy Module]{
	\begin{definition}
	The set of all syzygies among \(f_1,\dots,f_r\) is called the \emph{syzygy module} of \(f_1,\dots,f_r\), denoted
	\[
	\Syz(f_1,\dots,f_r)
	=
	\left\{
	(a_1,\dots,a_r)\in R^r \;\middle|\; a_1f_1+\cdots+a_rf_r=0
	\right\}.
	\]
\end{definition}}
\begin{remark}
\(\Syz(f_1,\dots,f_r)\) is an \(R\)-submodule of \(R^r\).
\begin{proof}
Suppose $f_1,\dots,f_r\in M$, where $M$ is a $R$-module. Then $\Syz(f_1,\dots,f_r)$ be a $R$-module: \begin{enumerate}[(i)]
	\item Let $(a_1,\dots,a_r),(b_1,\dots,b_r)\in\Syz(f_1,\dots,f_r)$. Since $\sum_{i=1}^r a_if_i=0$ and $\sum_{i=1}^r b_if_i=0$, then $\sum_{i=1}^r a_if_i+\sum_{i=1}^r b_if_i=\sum_{i=1}^r (a_i+b_i)f_i=0$. So the sum is again a syzygy.
	\item Let $c\in R$. Then $\sum_{i=1}^r(ca_i)f_i=c\sum_{i=1}^ra_if_i=c\dots 0=0$. So multiplying by an element of $\R$ gives another syzygy.
\end{enumerate}
Hence $\Syz(f_1,\dots,f_r)$ be a $R$-module.
\end{proof}
\end{remark}
			
\begin{example}
Consider \(R=\C[x,y]\). Then\[
\Syz(x,y)=\C[x,y]\cdot(-y,x).
\]
That is, every syzygy of \(x\) and \(y\) has the form
$(-gy,gx)$ for some \(g(x,y)\in \C[x,y]\).
\end{example}	
\begin{remark}
There are infinitely many syzygies, but they are all
generated by one basic syzygy.
\end{remark}

\section{Syzygies in Algebraic Geometry}

In algebraic geometry, one can study a projective variety by studying the equations that define it inside projective space. 
%Let
%\[
%X\subset \mathbb{P}^n
%\]
%be a projective variety, and let $
%I(X)\subset k[x_0,\dots,x_n]$ be its homogeneous ideal. If
%\[
%I(X)=(f_1,\dots,f_r),
%\]
%then the polynomials \(f_1,\dots,f_r\) are the equations cutting out \(X\).
%
%The equations themselves may satisfy relations. That is, there may exist polynomials\[
%a_1,\dots,a_r\in k[x_0,\dots,x_n]\quad\text{such that}\quad
%a_1f_1+\cdots+a_rf_r=0.
%\]
%Such a relation is called a \emph{syzygy} among the equations \(f_1,\dots,f_r\). Thus,
%\begin{itemize}
%	\item the equations \(f_1,\dots,f_r\) describe the variety \(X\),
%	\item the syzygies describe how those equations fit together.
%\end{itemize}
%So syzygies detect hidden algebraic structure inside the defining ideal of the variety. 
%%They tell us that the equations are not merely a list of unrelated polynomials, but rather a system whose members may depend on one another.
%%For this reason, syzygies are central in modern algebraic geometry: they provide finer information than the defining equations alone.

\medskip

\begin{example}
Consider the map \[
\phi:\mathbb{P}^1\to \mathbb{P}^3,
\qquad
[s:t]\mapsto [s^3:s^2t:st^2:t^3].
\]
Its image is called the \emph{twisted cubic}. We write the homogeneous coordinates on \(\mathbb{P}^3\) as
\[
[x_0:x_1:x_2:x_3],
\] Then points on the image satisfy 
\[
x_0=s^3,\qquad x_1=s^2t,\qquad x_2=st^2,\qquad x_3=t^3.
\] We immediately get the quadratic relations
\[
x_1^2=x_0x_2,\qquad x_1x_2=x_0x_3,\qquad x_2^2=x_1x_3.
\]
Thus the image lies on the three quadrics
\[
{\color{cyan}f_1=x_1^2-x_0x_2},\qquad
{\color{magenta}f_2=},\qquad
{\color{orange}f_3=x_2^2-x_1x_3}.
\]
These equations define the twisted cubic. However, the equations themselves are not independent. Indeed,
\begin{align*}
x_2{\color{cyan}f_1}-x_1{\color{magenta}f_2}+x_0{\color{orange}f_3}&=x_2{\color{cyan}(x_1^2-x_0x_2)}-x_1{\color{magenta}(x_1x_2-x_0x_3)}+x_0{\color{orange}(x_2^2-x_1x_3)}\\
&=x_1x_1^2-x_0x_2^2-x_1^2x_2+x_0x_1x_3+x_0x_2^2-x_0x_1x_3\\
&=0.
\end{align*}
So \((x_2,-x_1,x_0)\) is a syzygy among \(f_1,f_2,f_3\). Therefore,

\begin{itemize}
	\item the equations \(f_1,f_2,f_3\) describe the twisted cubic,
	\item the syzygy \((x_2,-x_1,x_0)\) describes how these equations fit together.
\end{itemize}
In algebraic geometry, the equations describe a variety, while syzygies describe relations among those equations.
\end{example}
