\noindent
A generic communication model has the form
\[
\vect{x}=(x_1,\dots,x_k)
\xrightarrow{\text{encoder}}
\underbrace{\vect{c}=(c_1,\dots,c_n)}_{\text{codeword}\; (n\geq k)}
\xrightarrow{\text{noise }\vect e}
\vect{y}=\vect{c}+\vect{e}
\xrightarrow{\text{decoder}}
\hat{\vect{x}}=(\hat x_1,\dots,\hat x_k).
\]

\[
\fullfunction{\mathsf{Encode}}{\Sigma^k}{\Sigma^n}{(x_1,\dots,x_k)}{(c_1,\dots,c_n)}

This is called the \emph{repetition code}. If one symbol is changed during transmission, we can still recover the original message by \emph{majority vote}. For instance,
\[
(0,0,0)\rightsquigarrow (0,1,0),
\qquad
(1,1,1)\rightsquigarrow (1,0,1).
\]
Since \(0\) appears most often in \((0,1,0)\), we decode it as \(0\), and since
\(1\) appears most often in \((1,0,1)\), we decode it as \(1\).

\defbox[Linear Encoding]{\begin{definition}

\begin{remark}
If $\dim_{\F_q}(\code)=k$, then \(\code\) is called a \textbf{linear \([n,k]\)-code over \(\F_q\)}.
\end{remark}

\probox[]{\begin{proposition}
If \(\code = \nkcode{n}{k}{q}\) is an linear code over \(\F_q\), then
\[
|\code|=q^k.
\]
\end{proposition}}
\begin{proof}
Choose a basis \(\vect b_1,\dots,\vect b_k\) of \(\code\). Every codeword can be written uniquely as
\[
x_1\vect b_1+\cdots+x_k\vect b_k,
\qquad x_i\in \F_q.
\]
Each \(x_i\) has \(q\) choices, and there are \(k\) independent coefficients. Therefore the number of codewords is \(q^k\).
\end{proof}

\begin{flushleft}\color{blue}
\begin{remark}[Linear codes as a short exact sequence]
	A linear \((n,k)\) code over the finite field \(\F_q\) can be organized by the short exact sequence
	\[
	0 \longrightarrow \F_q^k \xrightarrow{\;\mathsf{Encode}\;} \F_q^n \xrightarrow{\;\mathsf{Synd}\;} \F_q^{\,n-k} \longrightarrow 0.
	\]
	Here \(\mathsf{Encode}\) is an injective linear map, \(\mathsf{Synd}\) is a surjective linear map, and exactness means
	\[
	\operatorname{im}(\mathsf{Encode})=\ker(\mathsf{Synd}).
	\]
	Thus the code is
	\[
	\code=\operatorname{im}(\mathsf{Encode})=\ker(\mathsf{Synd})\subseteq \F_q^n.
	\]
	
	This viewpoint reformulates the usual ideas as follows.
	
	\begin{itemize}
		\item The code adds structured redundancy by embedding the message space \(\F_q^k\) into the larger ambient space \(\F_q^n\). The quotient
		\[
		\F_q^n/\code \cong \F_q^{\,n-k}
		\]
		measures the check information.
		
		\item Since \(\mathsf{Encode}\) is injective, one has
		\[
		\code \cong \F_q^k,
		\]
		and therefore
		\[
		|\code|=q^k.
		\]
		
		\item A generator matrix \(G\) is simply the matrix of the map
		\[
		\mathsf{Encode}:\F_q^k\to \F_q^n,
		\qquad
		\vect x \longmapsto \vect x G.
		\]
		
		\item A parity-check matrix \(H\) is the matrix of the map
		\[
		\mathsf{Synd}:\F_q^n\to \F_q^{\,n-k},
		\qquad
		\vect v \longmapsto H\vect v\transpose.
		\]
		Hence
		\[
		\code=\{\vect v\in \F_q^n : H\vect v\transpose=0\}.
		\]
	\end{itemize}
	
	If the generator matrix is in systematic form
	\[
	G=[I_k\mid A],
	\]
	then
	\[
	\mathsf{Encode}(\vect x)=(\vect x,\vect xA),
	\]
	so the message symbols appear directly as the first \(k\) coordinates, while the remaining \(n-k\) coordinates are parity symbols. In this case the short exact sequence splits:
	there is a projection
	\[
	p:\F_q^n\to \F_q^k
	\]
	onto the first \(k\) coordinates such that
	\[
	p\circ \mathsf{Encode}=\mathrm{id}_{\F_q^k}.
	\]
	Equivalently,
	\[
	\F_q^n \cong \F_q^k \oplus \F_q^{\,n-k},
	\]
	and the code is the graph of the linear map \(A:\F_q^k\to \F_q^{\,n-k}\).
	
	A compatible parity-check matrix is then
	\[
	H=[-A\transpose\mid I_{n-k}],
	\]
	since for every \(\vect x\in \F_q^k\),
	\[
	H(\vect xG)\transpose=0.
	\]
	So the code may be described equivalently as
	\[
	\code=\operatorname{im}(G)=\ker(H).
	\]
\end{remark}
\end{flushleft}
