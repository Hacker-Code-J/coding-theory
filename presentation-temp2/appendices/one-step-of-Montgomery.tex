\begin{algorithm}[H]
	\caption{One step of the Curve25519 Montgomery Ladder}
	\Fn{$\text{\normalfont ladderstep}\ (X_1,X_2,Z_2,X_3,Z_3)$}{
		$T_1\gets X_2 + Z_2$; $T_2\gets X_2 - Z_2$\;
		$T_7\gets T_2^2$; $T_6\gets T_1^2$\;
		$T_5\gets T_6-T_7$\;
		$T_3\gets X_3+Z_3$; $T_4\gets X_3-Z_3$\;
		$T_9\gets T_3\cdot T_2$; $T_8\gets T_4\cdot T_1$\;
		$X_3\gets T_8+T_9$; $Z_3\gets T_8-T_9$\;
		$X_3\gets X_3^2$; $Z_3\gets Z_3^2$\;
		$Z_3\gets Z_3\cdot X_1$\;
		$X_2\gets T_6\cdot T_7$\;
		$Z_2\gets 121666\cdot T_5$\tcp*{$121666 = (486662 + 2) / 4$}
		$Z_2\gets Z_2+T_7$\;
		$Z_2\gets Z_2\cdot T_5$\;
		\Return $(X_2,Z_2,X_3,Z_3)$\;
	}
\end{algorithm}
