	\draw[->] (-2,0) -- (2,0) node[right] {$x$};
	\draw[->] (0,-2) -- (0,2) node[above] {$y$};
	\fill[white,draw=black] (0,0) circle (1.5pt);
	\def\arrowlen{1}
	\pgfmathsetmacro{\mmin}{1/(sqrt(2)*1)}  % lower bound of m over the square
	\foreach \t in {0,10,...,350}{
		\pgfmathsetmacro\x{cos(\t)}
		\pgfmathsetmacro\y{sin(\t)}
		% On r=1, F = (-y,x)
		\pgfmathsetmacro\dx{(-\y)*\arrowlen}
		\pgfmathsetmacro\dy{(\x)*\arrowlen}
		
		\pgfmathsetmacro{\tval}{(1 - \mmin)/(1 - \mmin)}           % normalize
		\pgfmathsetmacro{\t}{min(1,max(0,\tval))}                     % clamp
		
		% "cool" colormap = linear from cyan (0,1,1) to magenta (1,0,1):
		%   R = t,  G = 1 - t,  B = 1
		\pgfmathsetmacro{\colR}{\t}
		\pgfmathsetmacro{\colG}{1-\t}
		\pgfmathsetmacro{\colB}{1}
		
		\draw[->, line width=.25mm,
		draw={rgb,1:red,\colR; green,\colG; blue,\colB}] 
		(\x,\y) -- ++(\dx,\dy);
	}
