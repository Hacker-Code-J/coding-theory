		% ==============================================================================

			\node[term label, text=blue] at (0, 1.3) {m = 1: $|GL_1| = \textcolor{blue}{(q-1)}$};

				\textbf{$q^1 - 1$} ways
			};

		\begin{scope}[shift={(1, 4.5)}, scale=1]

				\node[hl-span, fit=(M2_2-1-1)(M2_2-2-1)] {}; % v1 is forbidden
